Slideout Drawer

A jQuery plugin that creates sliding drawers for your site. Create them relative to any ancestor element or fixed to the window.

Syntax with Options:
<div class="your-container">
    <a class="icon-ellipsis-vertical icon-lg your-toggle-button" href="#"></a>

    <div class="slideout-drawer">
        <a class="slideout-drawer-close icon-remove" href="#"></a>
        <div class="slideout-drawer-content">
            ... Your Content ...
        </div>
    </div>
</div>
$('.your-toggle-button').slideoutDrawer({
    align: 'left', // left, right; position fixed also has top and bottom
    close: '.slideout-drawer-close', // class or id for close button
    content: '.slideout-drawer-content', // class or id for content element
    drawer: '.slideout-drawer', // class or id for drawer element
    relativeParent: '.your-container', // parent container the drawer should position and size to
    position: 'relative', // relative, fixed
    width: '75%' // px, %, em, rem
});
